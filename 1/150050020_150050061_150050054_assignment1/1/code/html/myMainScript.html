Shrinking image

tic;
img_1 = imread('../data/circles_concentric.png');
D = 2;
new_img_D2 = myShrinkImageByFactorD(img_1, 2);
new_img_D3 = myShrinkImageByFactorD(img_1, 3);

figure();
imagesc(img_1);
colormap (myColorScale);
title('Original Image');
daspect([1 1 1]);
colorbar;

figure();
imagesc(new_img_D2);
colormap (myColorScale);
title('D = 2');
daspect([1 1 1]);
colorbar;

figure();
imagesc(new_img_D3);
